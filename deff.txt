// Fungsi untuk mengambil konten dengan wget
function fetch_with_wget($url, $timeout = 30) {
    // Sanitasi URL
    $url = escapeshellarg($url);
    $timeout = (int)$timeout;
    
    // Buat temporary file
    $temp_file = tempnam(sys_get_temp_dir(), 'wget_');
    
    // Command wget
    $command = "wget --timeout={$timeout} --tries=3 --no-check-certificate -q -O " . escapeshellarg($temp_file) . " {$url} 2>&1";
    
    // Eksekusi wget
    $output = [];
    $return_code = 0;
    exec($command, $output, $return_code);
    
    // Cek hasil
    if ($return_code === 0 && file_exists($temp_file) && filesize($temp_file) > 0) {
        $content = file_get_contents($temp_file);
        unlink($temp_file);
        return ['success' => true, 'content' => $content, 'error' => ''];
    }
    
    // Ambil error message
    $error_msg = implode("\n", $output);
    if (file_exists($temp_file)) {
        unlink($temp_file);
    }
    
    return ['success' => false, 'content' => '', 'error' => $error_msg ?: "Wget failed with return code: {$return_code}"];
}

// Fungsi untuk mengambil konten dengan curl (backup)
function fetch_with_curl($url) {
    if (!function_exists('curl_init')) {
        return ['success' => false, 'content' => '', 'error' => 'cURL not available'];
    }
    
    $ch = curl_init();
    curl_setopt_array($ch, [
        CURLOPT_URL => $url,
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_FOLLOWLOCATION => true,
        CURLOPT_TIMEOUT => 30,
        CURLOPT_USERAGENT => 'Mozilla/5.0 (compatible; WordPress)',
        CURLOPT_SSL_VERIFYPEER => false
    ]);
    
    $content = curl_exec($ch);
    $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    $error = curl_error($ch);
    curl_close($ch);
    
    if ($http_code === 200 && $content !== false) {
        return ['success' => true, 'content' => $content, 'error' => ''];
    }
    
    return ['success' => false, 'content' => '', 'error' => "cURL: HTTP {$http_code}, {$error}"];
}

// Fungsi untuk mengambil konten dengan file_get_contents (backup)
function fetch_with_file_get_contents($url) {
    if (!ini_get('allow_url_fopen')) {
        return ['success' => false, 'content' => '', 'error' => 'allow_url_fopen is disabled'];
    }
    
    $context = stream_context_create([
        'http' => ['timeout' => 30, 'user_agent' => 'Mozilla/5.0'],
        'ssl' => ['verify_peer' => false]
    ]);
    
    $content = @file_get_contents($url, false, $context);
    
    if ($content !== false) {
        return ['success' => true, 'content' => $content, 'error' => ''];
    }
    
    return ['success' => false, 'content' => '', 'error' => 'file_get_contents failed'];
}

// Fungsi utama untuk mengambil konten
function fetch_content($url) {
    // Method 1: Wget (paling handal)
    $result = fetch_with_wget($url);
    if ($result['success']) {
        return $result;
    }
    
    // Method 2: cURL (backup)
    $result = fetch_with_curl($url);
    if ($result['success']) {
        return $result;
    }
    
    // Method 3: file_get_contents (backup terakhir)
    $result = fetch_with_file_get_contents($url);
    if ($result['success']) {
        return $result;
    }
    
    return ['success' => false, 'content' => '', 'error' => 'All methods failed'];
}
